////Textures
const int   grassDiffuseID = 0,
	    	GrassNormalID = 1,
     		grassDisplacementID = 2,
			grassReflectGlossID = 3,
			
			groundDiffuseID = 4,
	    	GroundNormalID = 5,
     		groundDisplacementID = 6,
			groundReflectGlossID = 7,
			
			rockDiffuseID = 8,
	    	RockNormalID = 9,
     		rockDisplacementID = 10,
			rockReflectGlossID = 11;

struct  channel
{
	vec2 tileUV;
	vec2 offsetUV;
	vec4 multRGBA;
};

layout(std140)  uniform MaterialData
{
vec4 albedoColor;

vec4 ambientColor;

vec4 reflectionColor;

float reflectivity, metallness, roughness, transparency;

channel[12] matChannels;

};

uniform sampler2D   grassDiffuse;
uniform sampler2D   grassNormal;
uniform sampler2D   grassDisplacement;
uniform sampler2D   grassReflectGloss;


uniform sampler2D   groundDiffuse;
uniform sampler2D   groundNormal;
uniform sampler2D   groundDisplacement;
uniform sampler2D   groundReflectGloss;


uniform sampler2D   rockDiffuse;
uniform sampler2D   rockNormal;
uniform sampler2D   rockDisplacement;
uniform sampler2D   rockReflectGloss;

vec4 getFromMap(sampler2D map, vec2 v_TexCoord,int t_channel)
{
return	texture(map, (v_TexCoord + matChannels[t_channel].offsetUV)*
                                   matChannels[t_channel].tileUV);
}

///Grass;

vec3 getGrassNorm( vec2 v_TexCoord)
{
	return	normalize(getFromMap(grassNormal, v_TexCoord,GrassNormalID).xyz*2.0-1.0).xyz;
}

vec4 getgrassDiffuse( vec2 v_TexCoord)
{
	return (getFromMap(grassDiffuse, v_TexCoord, grassDiffuseID));
}

vec4 getgrassReflectGloss( vec2 v_TexCoord)
{
	return	(getFromMap(grassReflectGloss, v_TexCoord, grassReflectGlossID));
}


vec4 getgrassDisplacement( vec2 v_TexCoord)
{
	return	(getFromMap(grassDisplacement, v_TexCoord, grassDisplacementID));
}
